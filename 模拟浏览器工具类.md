    public static DriverResponse download(SeleniumConfig seleniumConfig, String url) {
        LOGGER.info("开始下载页面, url: {}, seleniumConfig: {}", url, seleniumConfig);
        RemoteWebDriver driver = null;
        DriverResponse driverResponse = new DriverResponse();
        try {
            driver = createRemoteDriver(seleniumConfig);
            LOGGER.info("成功创建 RemoteWebDriver, 开始访问URL: {}", url);
            driver.get(url);
            if( seleniumConfig==null || StringUtils.isEmpty(seleniumConfig.getCheckBy()) ){
                driver.findElement(By.xpath("/html"));
                LOGGER.info("使用默认XPath检查页面加载");
            }else{
                driver.findElement(By.cssSelector(seleniumConfig.getCheckBy()));
                LOGGER.info("使用自定义选择器检查页面加载: {}", seleniumConfig.getCheckBy());
            }
            Thread.sleep(300);
            driverResponse.setPageSource(driver.getPageSource());
            driverResponse.setCookies(driver.manage().getCookies());
            LOGGER.info("页面下载完成, 页面源码长度: {}", driver.getPageSource().length());
        }catch (UnreachableBrowserException e){
            LOGGER.error("无法创建浏览器驱动", e);
            throw new TextException(TextExceptionEnum.NO_CREATE_DRIVER.getMessage());
        }catch(Exception e) {
            LOGGER.error("下载页面时发生异常", e);
            e.printStackTrace();
        }finally {
            if(driver!=null) {
                driver.quit();
                LOGGER.info("WebDriver 已关闭");
            }
        }
        return driverResponse;
    }

    /**
     * 新建RemoteWebDriver，用chrome，根据配置判断是调用远程还是本地
     * @param seleniumConfig
     * @return
     */
    public static RemoteWebDriver createDefaultDriver(SeleniumConfig seleniumConfig) {
        LOGGER.info("创建默认驱动, driverMode: {}", getSeleniumProperties().getDriverMode());
        if (getSeleniumProperties().getDriverMode() == DriverMode.REMOTE) {
            LOGGER.info("使用远程驱动模式");
            return createRemoteDriver(seleniumConfig);
        }
        if (getSeleniumProperties().getDriverMode() == DriverMode.LOCAL) {
            LOGGER.info("使用本地驱动模式");
            return createChromeDriver(seleniumConfig);
        }
        throw new TextException("缺少driver_mode配置!");
    }

    /**
     * 新建RemoteWebDriver，用chrome
     * @return
     * @throws MalformedURLException
     */
    public static RemoteWebDriver createRemoteDriver(SeleniumConfig seleniumConfig) {
        LOGGER.info("开始创建 RemoteWebDriver, remoteUrl: {}", seleniumProperties.getRemoteUrl());
        RemoteWebDriver driver = null;
        try {
            // 避免网站检测
            if (seleniumConfig.getIsUndetected()==null || seleniumConfig.getIsUndetected()) {
                LOGGER.info("使用防检测模式的 CdpRemoteWebDriver");
                ChromeOptions options = createChromeOptions(seleniumConfig);
                LOGGER.info("ChromeOptions 创建完成, 开始创建 CdpRemoteWebDriver");
                CdpRemoteWebDriver cdpDriver  = new CdpRemoteWebDriver(new URL(seleniumProperties.getRemoteUrl()), options);
                LOGGER.info("CdpRemoteWebDriver 创建成功, 开始执行反检测处理");
                UndetectedUtil.headless(cdpDriver);
                UndetectedUtil.cdcProps(cdpDriver);
                driver = cdpDriver;
                LOGGER.info("防检测 RemoteWebDriver 初始化完成");
            } else {
                LOGGER.info("使用标准 RemoteWebDriver");
                driver = new RemoteWebDriver(new URL(seleniumProperties.getRemoteUrl()), createChromeOptions(seleniumConfig));
                LOGGER.info("标准 RemoteWebDriver 创建成功");
            }
        } catch (MalformedURLException e) {
            LOGGER.error("RemoteURL格式错误: {}", seleniumProperties.getRemoteUrl(), e);
            e.printStackTrace();
        } catch (Exception e) {
            LOGGER.error("创建 RemoteWebDriver 时发生异常", e);
            throw new RuntimeException("创建 RemoteWebDriver 失败", e);
        }
        // 等待时间
        wait = new WebDriverWait(driver, Duration.ofSeconds(30));
        //设置定位时间
        driver.manage().timeouts().implicitlyWait(30, TimeUnit.SECONDS);
        //设置页面加载时间
        driver.manage().timeouts().pageLoadTimeout(60, TimeUnit.SECONDS);
        LOGGER.info("RemoteWebDriver 配置完成");
        return driver;
    }

    /**
     * 新建ChromeDriver
     * @return
     * @throws MalformedURLException
     */
    public static ChromeDriver createChromeDriver(SeleniumConfig seleniumConfig) {
        LOGGER.info("开始创建 ChromeDriver, localDriverPath: {}", seleniumProperties.getLocalDriverPath());

        // 先诊断环境
        diagnoseEnvironment();

        System.getProperties().setProperty("webdriver.chrome.driver", seleniumProperties.getLocalDriverPath());
        LOGGER.info("系统属性 webdriver.chrome.driver 已设置: {}", System.getProperty("webdriver.chrome.driver"));

        ChromeDriver driver;
        try {
            // 避免网站检测
            if (seleniumConfig.getIsUndetected()==null || seleniumConfig.getIsUndetected()) {
                LOGGER.info("使用防检测模式的 CdpChromeDriver");
                ChromeOptions options = createChromeOptions(seleniumConfig);
                LOGGER.info("开始创建 CdpChromeDriver");
                CdpChromeDriver cdpDriver  = new CdpChromeDriver(options);
                LOGGER.info("CdpChromeDriver 创建成功, 开始执行反检测处理");
                UndetectedUtil.headless(cdpDriver);
                UndetectedUtil.cdcProps(cdpDriver);
                driver = cdpDriver;
                LOGGER.info("防检测 ChromeDriver 初始化完成");
            } else {
                LOGGER.info("使用标准 ChromeDriver");
                driver = new ChromeDriver(createChromeOptions(seleniumConfig));
                LOGGER.info("标准 ChromeDriver 创建成功");
            }
        } catch (Exception e) {
            LOGGER.error("创建 ChromeDriver 时发生异常", e);
            // 尝试使用标准 ChromeDriver 作为备选方案
            LOGGER.info("尝试使用标准 ChromeDriver 作为备选方案");
            try {
                driver = new ChromeDriver(createChromeOptions(seleniumConfig));
                LOGGER.info("备选方案 ChromeDriver 创建成功");
            } catch (Exception e2) {
                LOGGER.error("备选方案也失败", e2);
                throw new RuntimeException("无法创建 ChromeDriver", e2);
            }
        }

        // 等待时间
        wait = new WebDriverWait(driver, Duration.ofSeconds(10));
        //设置定位时间
        driver.manage().timeouts().implicitlyWait(10, TimeUnit.SECONDS);
        //设置页面加载时间
        driver.manage().timeouts().pageLoadTimeout(20, TimeUnit.SECONDS);
        LOGGER.info("ChromeDriver 配置完成");
        return driver;
    }

    private static void diagnoseEnvironment() {
        LOGGER.info("=== 开始环境诊断 ===");
        try {
            // 检查 ChromeDriver 文件
            String chromeDriverPath = seleniumProperties.getLocalDriverPath();
            java.io.File chromeDriverFile = new java.io.File(chromeDriverPath);
            LOGGER.info("ChromeDriver 路径: {}", chromeDriverPath);
            LOGGER.info("ChromeDriver 文件存在: {}", chromeDriverFile.exists());
            LOGGER.info("ChromeDriver 可读: {}", chromeDriverFile.canRead());
            LOGGER.info("ChromeDriver 可执行: {}", chromeDriverFile.canExecute());
            LOGGER.info("ChromeDriver 文件大小: {} bytes", chromeDriverFile.length());

            // 检查 Chrome 浏览器
            String chromePath = "C:\\Users\\Administrator\\AppData\\Local\\Google\\Chrome\\Application\\chrome.exe";
            java.io.File chromeFile = new java.io.File(chromePath);
            LOGGER.info("Chrome 浏览器路径: {}", chromePath);
            LOGGER.info("Chrome 浏览器存在: {}", chromeFile.exists());
            LOGGER.info("Chrome 浏览器可执行: {}", chromeFile.canExecute());

            // 测试 ChromeDriver 命令行
            LOGGER.info("测试 ChromeDriver 命令行执行...");
            Process process = Runtime.getRuntime().exec(chromeDriverPath + " --version");
            java.io.BufferedReader reader = new java.io.BufferedReader(new java.io.InputStreamReader(process.getInputStream()));
            String line;
            while ((line = reader.readLine()) != null) {
                LOGGER.info("ChromeDriver 版本输出: {}", line);
            }
            int exitCode = process.waitFor();
            LOGGER.info("ChromeDriver 命令行退出码: {}", exitCode);

        } catch (Exception e) {
            LOGGER.error("环境诊断过程中发生异常", e);
        }
        LOGGER.info("=== 环境诊断结束 ===");
    }

    /**
     * 生成ChromeOptions
     * @param seleniumConfig
     * @return
     */
    public static ChromeOptions createChromeOptions(SeleniumConfig seleniumConfig) {
        LOGGER.info("开始创建 ChromeOptions, 配置: headless={}, loadImg={}, noSandbox={}, disableGPU={}, incognito={}",
                seleniumConfig.getIsHeadless(), seleniumConfig.getLoadImg(),
                seleniumConfig.getIsNoSandbox(), seleniumConfig.getIsDisableGPU(),
                seleniumConfig.getIsIncognito());

        ChromeOptions chromeOptions = new ChromeOptions();

        // 指定 Chrome 浏览器路径
        String chromePath = "C:\\Users\\Administrator\\AppData\\Local\\Google\\Chrome\\Application\\chrome.exe";
        chromeOptions.setBinary(chromePath);
        LOGGER.info("设置 Chrome 二进制路径: {}", chromePath);

        // 无界面参数
        if( seleniumConfig.getIsHeadless()==null || seleniumConfig.getIsHeadless() ) {
            chromeOptions.setHeadless(true);
            // 窗体最大化
            chromeOptions.addArguments("start-maximized");
            LOGGER.info("启用无头模式");
        }
        // 不加载图片
        if( seleniumConfig.getLoadImg()==null || !seleniumConfig.getLoadImg() ){
            chromeOptions.addArguments("blink-settings=imagesEnabled=false");
            LOGGER.info("禁用图片加载");
        }
        // 不打开sandbox
        if( seleniumConfig.getIsNoSandbox()==null || seleniumConfig.getIsNoSandbox() ){
            chromeOptions.addArguments("--no-sandbox");
            LOGGER.info("禁用沙盒模式");
        }
        // 禁止GPU
        if( seleniumConfig.getIsDisableGPU()==null || seleniumConfig.getIsDisableGPU() ){
            chromeOptions.addArguments("--disable-gpu");
            LOGGER.info("禁用GPU加速");
        }
        // 无痕模式
        if( seleniumConfig.getIsIncognito()!=null && seleniumConfig.getIsIncognito() ){
            chromeOptions.addArguments("--incognito");
            LOGGER.info("启用无痕模式");
        }
        // 添加重要参数
        chromeOptions.addArguments("--disable-dev-shm-usage");
        chromeOptions.addArguments("--disable-blink-features=AutomationControlled");
        chromeOptions.setExperimentalOption("excludeSwitches", Collections.singletonList("enable-automation"));
        chromeOptions.setExperimentalOption("useAutomationExtension", false);
        // 设置为中文（有界面才生效）
        chromeOptions.setExperimentalOption("prefs", Collections.singletonMap("intl.accept_languages", "zh-CN,zh;q=0.9"));
        LOGGER.info("设置浏览器语言为中文");

        //设置文件下载到指定目录
        String downloadFilepath = seleniumProperties.getDownloadPath().replace("\\\\","\\");
        chromeOptions.setCapability(CapabilityType.ACCEPT_INSECURE_CERTS, true);
        HashMap<String, Object> chromePrefs = new HashMap<String, Object>();
        chromePrefs.put("profile.default_content_settings.popups", 0);
        chromePrefs.put("download.default_directory", downloadFilepath);
        chromeOptions.setExperimentalOption("prefs", chromePrefs);
        chromeOptions.addArguments("--test-type");
        LOGGER.info("设置下载目录: {}", downloadFilepath);

        LOGGER.info("ChromeOptions 创建完成");
        return chromeOptions;
    }

    public static By parseBy(String key, String value){
        if(StringUtils.isEmpty(key) || StringUtils.isEmpty(value) ) return null;
        switch (key){
            case "id":
                return By.id(value);
            case "name":
                return By.name(value);
            case "xpath":
                return By.xpath(value);
            case "className":
                return By.className(value);
            case "linkText":
                return By.linkText(value);
            case "partialLinkText":
                return By.partialLinkText(value);
            case "tagName":
                return By.tagName(value);
            case "cssSelector":
                return By.cssSelector(value);
            default:
                return null;
        }
    }

    /**
     * 找标签
     * 未找到标签，抛出对应说明
     * @param elementName 标签说明
     * @param by
     * @param driver
     */
    public static WebElement findElement(String elementName, By by, WebDriver driver) {
        try{
            // 等待标签出现
            waitForElementToBeVisible(by);
            return driver.findElement(by);
        }catch(NoSuchElementException e){
            throw new TextException(elementName+"：对应的标签未找到");
        }
    }

    /**
     * 找标签并点击
     * 未找到标签，抛出对应说明
     * @param elementName 标签说明
     * @param by
     * @param driver
     */
    public static WebElement findElementAndClick(String elementName, By by, WebDriver driver) {
        try{
            WebElement element = findElement(elementName, by, driver);
            // 等待标签出现
            waitForElementToBeVisible(by);
            // 等待可以点击
            waitForElementToBeClickable(by);
            // 点击
            element.click();
            return element;
        }catch(TimeoutException e){
            throw new TextException(elementName+"：对应的标签不能点击");
        }
    }
